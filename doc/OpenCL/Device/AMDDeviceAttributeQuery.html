def topology_amd
  ptr1 = MemoryPointer::new( AMDTopology )
  error = OpenCL.clGetDeviceInfo(self, TOPOLOGY_AMD, ptr1.size, ptr1, nil)
  error_check(error)
  return AMDTopology::new(ptr1)
end
