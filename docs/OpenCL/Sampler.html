def context
  return @_context if @_context
  ptr = MemoryPointer::new( Context )
  error = OpenCL.clGetSamplerInfo(self, CONTEXT, Context.size, ptr, nil)
  error_check(error)
  @_context = Context::new( ptr.read_pointer )
end
