def self.make(major, minor = 0, patch = 0)
  ((major & MAJOR_MASK) << (MINOR_BITS + PATCH_BITS)) +
  ((minor & MINOR_MASK) << PATCH_BITS) +
   (patch & PATCH_MASK)
end

def <=>(other)
  res = (@major <=> other.major)
  res = (@minor <=> other.minor) if res == 0
  res = (@patch <=> other.patch) if res == 0
  res
end
