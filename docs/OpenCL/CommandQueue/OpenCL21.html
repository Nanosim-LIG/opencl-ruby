def device_default
  ptr = MemoryPointer::new( CommandQueue )
  error = OpenCL.clGetCommandQueueInfo(self, DEVICE_DEFAULT, CommandQueue.size, ptr, nil)
  error_check(error)
  return CommandQueue::new( ptr.read_pointer )
end
