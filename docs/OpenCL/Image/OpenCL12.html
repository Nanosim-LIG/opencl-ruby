def buffer
  ptr = MemoryPointer::new( Buffer )
  error = OpenCL.clGetImageInfo(self,  BUFFER, Buffer.size, ptr, nil)
  error_check(error)
  return nil if ptr.null?
  return Buffer::new(ptr.read_pointer)
end
