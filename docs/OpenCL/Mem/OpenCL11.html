def associated_memobject
  ptr = MemoryPointer::new( Mem )
  error = OpenCL.clGetMemObjectInfo(self, ASSOCIATED_MEMOBJECT, Mem.size, ptr, nil)
  error_check(error)
  return nil if ptr.read_pointer.null?
  return Mem::new( ptr.read_pointer )
end
