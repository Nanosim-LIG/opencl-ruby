def from_string(uuid)
  m = uuid.match(/(\h\h)(\h\h)(\h\h)(\h\h)(\h\h)(\h\h)(\h\h)(\h\h)/)
  raise "invalid format" unless m
  LUID_SIZE_KHR.times { |i|
    self[:id][LUID_SIZE_KHR-1-i] = m[i+1].to_i(16)
  }
  self
end

def to_s
