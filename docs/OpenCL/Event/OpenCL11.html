def context
  @_context ||= begin
    ptr = MemoryPointer::new( Context )
    error = OpenCL.clGetEventInfo(self, CONTEXT, Context.size, ptr, nil)
    error_check(error)
    Context::new( ptr.read_pointer )
  end
end
